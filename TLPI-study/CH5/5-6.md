# 演習5-6
```c
/* fd1でファイルオープン(すでにファイルがあったら初期化) */
fd1 = open(file, O_RDWR | O_CREAT | O_TRUNC, S_IRUSR | S_IWUSR);
/* dup()でfd1とfd2はオープンファイルテーブルを共有 */
fd2 = dup(fd1);
/* open()でfd3はi-node(ファイル実体)のみ共有 */
fd3 = open(file, O_RDWR);
/* 
 * fd1とfd2はオープンファイルテーブルの同じエントリを共有しているのでファイルオフセットも共有している。
 * よってファイルには"Hello, world"と書き込まれる
 */
write(fd1, "Hello,", 6);
write(fd2, " world", 6);
/* fd1とfd2のファイルオフセットがファイル先頭になる */
lseek(fd2, 0, SEEK_SET);
/* 先頭の"Hello"が上書きされて"HELLO"になる */
write(fd1, "HELLO", 6);
/* fd3のファイルオフセットはずっとファイル先頭のままなので、"HELLO,"が"Gidday"に上書きされる */
write(fd3, "Gidday", 6);
/* 最後にファイルには"Gidday world"が残る */
```
